        function getRandomDarkColor() {

            <input type="number" id="manualInput" class="input-field" min="0" placeholder="Set #">

            <button class="control-btn" id="setButton">✔ Set</button>

        </div>

        <div class="controls">

            <button class="control-btn" id="decreaseButton">⬅ Back</button>

            <button class="control-btn" id="resetButton">🔄 Reset</button>

        </div>

    <script>

        let count = 0;

        const callButton = document.getElementById('callButton');

        const decreaseButton = document.getElementById('decreaseButton');

        const resetButton = document.getElementById('resetButton');

        const setButton = document.getElementById('setButton');

        const manualInput = document.getElementById('manualInput');

        

        callButton.addEventListener('click', () => {

            count++;

            callButton.textContent = count;

        });

        

        decreaseButton.addEventListener('click', () => {

            if (count > 0) {

                count--;

                callButton.textContent = count;

            }

        });

        

        resetButton.addEventListener('click', () => {

            count = 0;

            callButton.textContent = count;

        });

        

        setButton.addEventListener('click', () => {

            const inputValue = parseInt(manualInput.value, 10);

            if (!isNaN(inputValue) && inputValue >= 0) {

                count = inputValue;

                callButton.textContent = count;

            }

        });

    </script>
